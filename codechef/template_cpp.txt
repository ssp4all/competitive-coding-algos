#include <bits/stdc++.h>
using namespace std;
#define endl "\n" 
#define lld long long int

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    lld n, x;
    cin>>n;
    vector<int> v;
    for(lld i=0; i<n; i++){
        cin >> x;
        v.push_back(x);
    }
    sort(v.begin(), v.end());
    for(auto ele: v)
        cout<<ele<<" ";
  return 0;
}
