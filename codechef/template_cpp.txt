#include <bits/stdc++.h>
using namespace std;
#define endl "\n" 
#define lld long long int

inline int scan(){
    int x = 0;
    char c = getchar_unlocked();
    bool b = 0;
    while ( c < '0' || c > '9'){
        if( c == '-'){
            b = 1;
        }
        c = getchar_unlocked();
    }
    while ( c >= '0' && c <= '9'){
        x = ( x << 3) + ( x << 1) + c - '0';
        c = getchar_unlocked();
    }
    if(b)
        return -x;
        
    return x;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    lld n, x;
    cin>>n;
    vector<int> v;
    for(lld i=0; i<n; i++){
        cin >> x;
        v.push_back(x);
    }
    sort(v.begin(), v.end());
    for(auto ele: v)
        cout<<ele<<" ";
  return 0;
}
